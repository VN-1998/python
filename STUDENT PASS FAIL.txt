class college:
    col= "NITTE"#ITS A CLASS VARIABLE ITS COMMMON TO ALL STUDENTS
    def student(self,name,mark):
        self.name = name
        self.mark = mark
    def passfail(self):
        if(self.mark > 40):
            print("Clear")
        else:
            print("Fail")
    def modify(self,grace):
        self.mark=self.mark+grace
    def display(self):
        print("Name:",self.name)
        print("College:",self.col)
        print("Mark:",self.mark)
        print("Status:",self.passfail())
